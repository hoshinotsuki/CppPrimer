    int main()
    {
		int w = 0; //初始质量
		int h = 0; //初试高度
		int w1 = 0;
		int h1 = 0;
		int w2 = 0;
		int h2 = 0;
		vector<int> va, vb, vc;

		string str[3], word;
		for (int i = 0; i<3; i++)
			getline(cin, str[i]);
		istringstream record(str[0]);
		while (record >> word)
			va.push_back(atoi(word.c_str()));
		w = va[0];
		h = va[1];

		word = "";
		istringstream record2(str[1]);
		while (record2 >> word)
			vb.push_back(atoi(word.c_str()));
		w1 = vb[0];
		h1 = vb[1];

		word = "";
		istringstream record3(str[2]);
		while (record3 >> word)
			vc.push_back(atoi(word.c_str()));
		w2 = vc[0];
		h2 = vc[1];

		int wnow = w;
		int hnow = h;
		while (hnow > 0)
		{
			wnow += hnow;
			if (hnow == h1)
			{
				wnow -= w1;
				if (wnow < 0)
					wnow = 0;
			}
			else
				if (hnow == h2)
				{
					wnow -= w2;
					if (wnow < 0)
						wnow = 0;
				}
			hnow--;
		}

		cout << wnow;

		return 0;
    }

	}
	else {
		while (!data.empty())
		{
			int n = (data.size() - 1) / 2;
