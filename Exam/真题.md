// 本题为考试单行多行输入输出规范示例，无需提交，不计分。
#include <iostream>
#include <string>
#include <vector>
using namespace std;

bool helper(int &sum, int & k,int i ,const int sign, vector<string> &ret)
{

	if (k == 0 && sum != 0)
	{
 
		ret.pop_back();
	 
		return false;
	}//步数用完
	
	if (k == 0 && sum == 0)
		return true;
	k--;
	if (sign == 1)
	{
		sum -= i;
		ret.push_back(to_string(i) + " " + "+");

		if (helper(sum, k, i * 2, 1, ret))
			return true;//加左子树
		k++;
		sum += i * 2;
		if (helper(sum, k, i * 2, 0, ret))
			return true;//减左子树
		k++;
		sum -=i * 2;
		
		if (helper(sum, k, i * 2 + 1, 1, ret))
			return true;//加右子树
		k++;
		sum += (i * 2+1);

		if (helper(sum, k, i * 2 + 1, 0, ret))
			return true;//减右子树
		k++;
		sum -= (i * 2 + 1);

	}
	else
		{
			sum += i;
			ret.push_back(to_string(i) + " " + "-");

			if (helper(sum, k, i * 2, 1, ret))
				return true;//加左子树
			k++;
			sum += i * 2;
			if (helper(sum, k, i * 2, 0, ret))
				return true;//减左子树
			k++;
			sum -= i * 2;

			if (helper(sum, k, i * 2 + 1, 1, ret))
				return true;//加右子树
			k++;
			sum += (i * 2 + 1);

			if (helper(sum, k, i * 2 + 1, 0, ret))
				return true;//减右子树
			k++;
			sum -= (i * 2 + 1);
		} 
}
